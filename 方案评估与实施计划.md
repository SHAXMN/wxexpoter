### 2. 问题根源

❌ **消息内容是压缩的 protobuf 格式**
- 微信 8.0+ 将消息存储为压缩的二进制数据
- MessageParser 假设消息是纯文本 XML
- 没有调用 RawMessage 来解析消息内容

❌ **缺少消息 protobuf 定义**
- RawMessage 需要知道字段编号（如 "1.2.3"）
- 微信消息的 protobuf 结构未知

```cpp
// 新增：扫描所有文档附件
bool ITunesDb::scanDocumentAttachments(
    const std::string& outputPath,
    std::vector<AttachmentInfo>& attachments) const
{
    sqlite3_stmt* stmt = NULL;
    std::string sql = "SELECT fileID, relativePath FROM Files "
                     "WHERE relativePath LIKE '%OpenData%' "
                     "AND (relativePath LIKE '%.pdf' "
                     "OR relativePath LIKE '%.doc' "
                     "OR relativePath LIKE '%.docx' "
                     "OR relativePath LIKE '%.xls' "
                     "OR relativePath LIKE '%.xlsx' "
                     "OR relativePath LIKE '%.ppt' "
                     "OR relativePath LIKE '%.pptx' "
                     "OR relativePath LIKE '%.zip' "
                     "OR relativePath LIKE '%.rar')";
    
    int rc = sqlite3_prepare_v2(m_db, sql.c_str(), -1, &stmt, NULL);
    if (rc != SQLITE_OK)
    {
        return false;
    }
    
    while (sqlite3_step(stmt) == SQLITE_ROW)
    {
        std::string fileId = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 0));
        std::string relativePath = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 1));
        
        AttachmentInfo info;
        info.fileId = fileId;
        info.relativePath = relativePath;
        
        // 提取会话 hash
        // relativePath 格式: Documents/.../OpenData/{hash}/{msgId}.ext
        size_t pos1 = relativePath.find("OpenData/");
        if (pos1 != std::string::npos)
        {
            pos1 += 9; // "OpenData/" 长度
            size_t pos2 = relativePath.find('/', pos1);
            if (pos2 != std::string::npos)
            {
                info.sessionHash = relativePath.substr(pos1, pos2 - pos1);
            }
        }
        
        attachments.push_back(info);
    }
    
    sqlite3_finalize(stmt);
    return true;
}

// 新增：批量复制附件
bool ITunesDb::copyAttachments(
    const std::vector<AttachmentInfo>& attachments,
    const std::string& outputPath) const
{
    int copied = 0;
    for (const auto& att : attachments)
    {
        // 构建源文件路径
        std::string srcPath = combinePath(m_backupPath, 
                                          att.fileId.substr(0, 2), 
                                          att.fileId);
        
        // 构建目标路径
        std::string filename = getFileName(att.relativePath);
        std::string destDir = combinePath(outputPath, 
                                          "attachments", 
                                          att.sessionHash);
        
        if (!existsDirectory(destDir))
        {
            makeDirectory(destDir);
        }
        
        std::string destPath = combinePath(destDir, filename);
        
        // 复制文件
        if (copyFile(srcPath, destPath))
        {
            copied++;
        }
    }
    
    return copied > 0;
}
```
